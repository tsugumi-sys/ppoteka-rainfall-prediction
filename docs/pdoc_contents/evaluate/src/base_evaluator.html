 92    def rescale_pred_tensor(self, target_tensor: torch.Tensor, target_param: str) -> torch.Tensor:
 93        """This function rescale target_tensor to target parameter's original scale.
 94        e.g. rain tensor [0, 1] -> [0, 100]
 95        """
 96        if target_tensor.max().item() > 1 or target_tensor.min().item() < 0:
 97            raise ValueError(
 98                "Invalid scale of target tensor "
 99                f"(max: {target_tensor.max().item()}, min: {target_tensor.min().item()}). Should be scaled to [0, 1]"
100            )
101
102        target_tensor = normalize_tensor(target_tensor, DEVICE)
103        min_val, max_val = MinMaxScalingValue.get_minmax_values_by_weather_param(target_param)
104        rescaled_tensor = rescale_tensor(min_value=min_val, max_value=max_val, tensor=target_tensor.cpu().detach())
105        return rescaled_tensor

304        """This function generates prediction dataframe from prediction tensor.
305
306        Return (pd.DataFrame): A prediction dataframe (columns: [`Pred_Value`], index: obsevation points name).
307        """
308        if pred_tensor.ndim > 2:
309            raise ValueError(
310                f"Invalid tensor dimentions for pred_tensor. The shape shold be less than 2, but {pred_tensor.shape}."

314            pred_ob_point_tensor = get_ob_point_values_from_tensor(pred_tensor, self.observation_point_file_path)

316            pred_ob_point_tensor = pred_tensor
317        else:
318            raise ValueError(f"Invalid predict tensor shape.: {pred_tensor.shape}")
319
320        with open(self.observation_point_file_path, "r") as f:
321            ob_point_data = json.load(f)

323        pred_df = pd.DataFrame(index=list(ob_point_data.keys()))
324        pred_df["Pred_Value"] = (
325            pred_ob_point_tensor.clone().detach().numpy().copy()
326        )  # torch.float convert to numpy.float32
327        return pred_df

429    def update_input_tensor(
430        self, before_input_tensor: torch.Tensor, before_standarized_info: Dict, next_frame_tensor: torch.Tensor
431    ) -> Tuple[torch.Tensor, Dict[str, float]]:
432        """Update input tensor X_test for next prediction. A next one frame tensor (prediction or label tensor) a given
433        and update the first frame of X_test with that given tensor.
434
435        Args:
436            before_input_tensor (torch.Tensor):
437            before_standarized_info (Dict): Stadarization information to rescale tensor using this.
438            next_frame_tensor (torch.Tensor): This tensor should be scaled to [0, 1].
439        """
440        if next_frame_tensor.max().item() > 1 or next_frame_tensor.min().item() < 0:
441            raise ValueError(
442                "next_frame_tensor is not scaled to [0, 1], "
443                f"but [{next_frame_tensor.min().item(), next_frame_tensor.max().item()}]"
444            )
445
446        # Convert next_frame tensor to grids if ob point tensor is given.
447        _, num_channels, _, height, width = before_input_tensor.size()
448        if next_frame_tensor.ndim == 2:
449            # The case of next_frame_tensor is [ob_point values]
450            _next_frame_tensor = next_frame_tensor.cpu().detach()
451            _next_frame_tensor = normalize_tensor(_next_frame_tensor, device="cpu")
452            _next_frame_ndarray = _next_frame_tensor.numpy().copy()
453            next_frame_tensor = torch.zeros(
454                (len(self.input_parameter_names), width, height), dtype=torch.float, device=DEVICE
455            )
456            for param_dim, weather_param in enumerate(self.input_parameter_names):
457                interpolator_interactor = InterpolatorInteractor()
458                interp_next_frame_ndarray = interpolator_interactor.interpolate(
459                    weather_param, _next_frame_ndarray[param_dim, ...], self.observation_point_file_path
460                )
461                next_frame_tensor[param_dim, ...] = torch.from_numpy(interp_next_frame_ndarray.copy()).to(DEVICE)
462            next_frame_tensor = normalize_tensor(next_frame_tensor, device=DEVICE)
463
464        scaling_method = self.hydra_cfg.scaling_method
465
466        if scaling_method == ScalingMethod.MinMax.value:
467            return (
468                torch.cat(
469                    (
470                        before_input_tensor[:, :, 1:, ...],
471                        torch.reshape(next_frame_tensor, (1, num_channels, 1, height, width)),
472                    ),
473                    dim=2,
474                ),
475                {},
476            )
477
478        # elif scaling_method == ScalingMethod.Standard.value or scaling_method == ScalingMethod.MinMaxStandard.value:
479        else:
480            for param_dim, param_name in enumerate(self.input_parameter_names):
481                means, stds = before_standarized_info[param_name]["mean"], before_standarized_info[param_name]["std"]
482                before_input_tensor[:, param_dim, ...] = before_input_tensor[:, param_dim, ...] * stds + means
483
484            updated_input_tensor = torch.cat(
485                (
486                    before_input_tensor[:, :, 1:, ...],
487                    torch.reshape(next_frame_tensor, (1, num_channels, 1, *before_input_tensor.size()[3:])),
488                ),
489                dim=2,
490            )
491            standarized_info = {}
492            for param_dim, param_name in enumerate(self.input_parameter_names):
493                standarized_info[param_name] = {}
494                means = torch.mean(updated_input_tensor[:, param_dim, ...])
495                stds = torch.std(updated_input_tensor[:, param_dim, ...])
496                updated_input_tensor[:, param_dim, ...] = (updated_input_tensor[:, param_dim, ...] - means) / stds
497                standarized_info[param_name]["mean"] = means
498                standarized_info[param_name]["std"] = stds
499            return updated_input_tensor, standarized_info

431        self, before_input_tensor: torch.Tensor, before_standarized_info: Dict, next_frame_tensor: torch.Tensor
432    ) -> Tuple[torch.Tensor, Dict[str, float]]:
433        """Update input tensor X_test for next prediction. A next one frame tensor (prediction or label tensor) a given
434        and update the first frame of X_test with that given tensor.
435
436        Args:
437            before_input_tensor (torch.Tensor):
438            before_standarized_info (Dict): Stadarization information to rescale tensor using this.
439            next_frame_tensor (torch.Tensor): This tensor should be scaled to [0, 1].
440        """
441        if next_frame_tensor.max().item() > 1 or next_frame_tensor.min().item() < 0:
442            raise ValueError(
443                "next_frame_tensor is not scaled to [0, 1], "
444                f"but [{next_frame_tensor.min().item(), next_frame_tensor.max().item()}]"
445            )
446
447        # Convert next_frame tensor to grids if ob point tensor is given.
448        _, num_channels, _, height, width = before_input_tensor.size()
449        if next_frame_tensor.ndim == 2:
450            # The case of next_frame_tensor is [ob_point values]
451            _next_frame_tensor = next_frame_tensor.cpu().detach()
452            _next_frame_tensor = normalize_tensor(_next_frame_tensor, device="cpu")
453            _next_frame_ndarray = _next_frame_tensor.numpy().copy()
454            next_frame_tensor = torch.zeros(
455                (len(self.input_parameter_names), width, height), dtype=torch.float, device=DEVICE
456            )
457            for param_dim, weather_param in enumerate(self.input_parameter_names):
458                interpolator_interactor = InterpolatorInteractor()
459                interp_next_frame_ndarray = interpolator_interactor.interpolate(
460                    weather_param, _next_frame_ndarray[param_dim, ...], self.observation_point_file_path
461                )
462                next_frame_tensor[param_dim, ...] = torch.from_numpy(interp_next_frame_ndarray.copy()).to(DEVICE)
463            next_frame_tensor = normalize_tensor(next_frame_tensor, device=DEVICE)
464
465        scaling_method = self.hydra_cfg.scaling_method
466
467        if scaling_method == ScalingMethod.MinMax.value:
468            return (
469                torch.cat(
470                    (
471                        before_input_tensor[:, :, 1:, ...],
472                        torch.reshape(next_frame_tensor, (1, num_channels, 1, height, width)),
473                    ),
474                    dim=2,
475                ),
476                {},
477            )
478
479        # elif scaling_method == ScalingMethod.Standard.value or scaling_method == ScalingMethod.MinMaxStandard.value:
480        else:
481            for param_dim, param_name in enumerate(self.input_parameter_names):
482                means, stds = before_standarized_info[param_name]["mean"], before_standarized_info[param_name]["std"]
483                before_input_tensor[:, param_dim, ...] = before_input_tensor[:, param_dim, ...] * stds + means
484
485            updated_input_tensor = torch.cat(
486                (
487                    before_input_tensor[:, :, 1:, ...],
488                    torch.reshape(next_frame_tensor, (1, num_channels, 1, *before_input_tensor.size()[3:])),
489                ),
490                dim=2,
491            )
492            standarized_info = {}
493            for param_dim, param_name in enumerate(self.input_parameter_names):

498                standarized_info[param_name]["mean"] = means
499                standarized_info[param_name]["std"] = stds
